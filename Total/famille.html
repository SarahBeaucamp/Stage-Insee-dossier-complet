function classer_familles(label) {
  if (!label) return null;
  let txt = label.toLowerCase();
  
  if (txt.includes("père seul")) return "Hommes seuls avec enfant(s)";
  if (txt.includes("mère seule")) return "Femmes seules avec enfant(s)";
  if (txt.includes("couple sans enfant")) return "Couple sans enfant(s)";
  if (txt.includes("couple avec enfant")) return "Couple avec enfant(s)";
  
  return null; // Ignore le reste
}

equip_mutate1 = equip_brut1.map(function(d) {
  return {
    OBS_VALUE: d.OBS_VALUE,
    Situation_f: classer_familles(d.TAB_MEASURE_LABEL)
  };
}).filter(d => d.Situation_f !== null);

equip_groupes1 = d3.rollups(
  equip_mutate1,
  function(v) { return d3.sum(v, function(d) { return d.OBS_VALUE; }); },
  function(d) { return d.Situation_f; }
);

equip_summarise1 = equip_groupes1.map(function(g) {
  return { Situation_f: g[0], total_groupe: g[1] };
});

total1 = d3.sum(equip_summarise1, function(d) { return d.total_groupe; });

equip1 = equip_summarise1.map(function(d) {
  return {
    Situation_f: d.Situation_f,
    total_groupe: d.total_groupe,
    total1: total1,
    part_groupe: total1 > 0 ? (d.total_groupe / total1) * 100 : 0
  };
});
camembert1 = {
  if (!isGeoSelected || total1 === 0) return htl.html`<div>Sélectionnez un territoire ou aucune donnée disponible pour cette année.</div>`;
  
  const width = 600;
  const height = 350;
  const radius = Math.min(width, height) / 2 - 20;

  const arc = d3.arc().innerRadius(0).outerRadius(radius);
  const pie = d3.pie().value(d => d.total_groupe).sort(null);

  const color = d3.scaleOrdinal()
    .domain([
      "Couple avec enfant(s)", 
      "Couple sans enfant(s)", 
      "Femmes seules avec enfant(s)", 
      "Hommes seuls avec enfant(s)"
    ])
    .range(["#B5E1FF","#86C6FF", "#268DFF", "#145CBF"]);

  const svg = d3.create("svg")
    .attr("viewBox", [0, 0, width, height])
    .style("font-family", "sans-serif");

  const gPie = svg.append("g")
    .attr("transform", `translate(${width / 3}, ${height / 2})`);

  // Tranches du camembert
  gPie.selectAll("path")
    .data(pie(equip1))
    .join("path")
    .attr("fill", d => color(d.data.Situation_f))
    .attr("stroke", "white")
    .attr("stroke-width", "1.5px")
    .attr("d", arc)
    .append("title")
    .text(d => `${d.data.Situation_f}: ${d.data.total_groupe}`);

  // Étiquettes de pourcentages au centre des tranches
  gPie.selectAll("text")
    .data(pie(equip1))
    .join("text")
    .attr("transform", d => `translate(${arc.centroid(d)})`)
    .attr("text-anchor", "middle")
    .attr("fill", d => d.data.part_groupe < 15 ? "black" : "white") // Texte noir si tranche très claire/petite
    .attr("font-size", "12px")
    .attr("font-weight", "bold")
    // Affiche uniquement si c'est assez grand pour ne pas baver (ex: > 2%)
    .text(d => d.data.part_groupe >= 2 ? `${d.data.part_groupe.toFixed(1)}%` : "");

  // Légende positionnée à droite
  const legend = svg.append("g")
    .attr("transform", `translate(${width * 0.65}, ${height / 2 - 60})`);
    
  const categories = color.domain();
  categories.forEach((cat, i) => {
    const row = legend.append("g").attr("transform", `translate(0, ${i * 25})`);
    row.append("rect").attr("width", 15).attr("height", 15).attr("fill", color(cat)).attr("rx", 2);
    row.append("text").attr("x", 25).attr("y", 12).attr("font-size", "12px").text(cat);
  });

  return svg.node();
}

function classer_enfants(label) {
  if (!label) return null;
  let txt = label.toLowerCase();
  
  if (txt.includes("aucun enfant")) return "Aucun enfant";
  if (txt.includes("1 enfant")) return "1 enfant";
  if (txt.includes("2 enfants")) return "2 enfants";
  if (txt.includes("3 enfants")) return "3 enfants";
  if (txt.includes("4 enfants") || txt.includes("plus")) return "4 enfants ou plus";
  
  return null; // Ignore le reste
}

equip_mutate = equip_brut.map(function(d) {
  return {
    OBS_VALUE: d.OBS_VALUE,
    enfants: classer_enfants(d.TAB_MEASURE_LABEL)
  };
}).filter(d => d.enfants !== null);

equip_groupes = d3.rollups(
  equip_mutate,
  function(v) { return d3.sum(v, function(d) { return d.OBS_VALUE; }); },
  function(d) { return d.enfants; }
);

equip_summarise = equip_groupes.map(function(g) {
  return { enfants: g[0], total_groupe: g[1] };
});

total = d3.sum(equip_summarise, function(d) { return d.total_groupe; });

equip = equip_summarise.map(function(d) {
  return {
    enfants: d.enfants,
    total_groupe: d.total_groupe,
    total: total,
    part_groupe: total > 0 ? (d.total_groupe / total) * 100 : 0
  };
});

treemap_chart = {
  if (!isGeoSelected || total === 0) return htl.html`<div>Sélectionnez un territoire ou aucune donnée disponible pour cette année.</div>`;

  const width = 600;
  const height = 350;

  const color = color_enfants;

  // Hiérarchie 
  const root = d3.hierarchy({ children: equip_trie })
    .sum(d => d.total_groupe)
    .sort((a, b) => b.value - a.value);

  d3.treemap()
    .size([width, height])
    .paddingInner(2)
    .paddingOuter(2)
    .round(true)
    (root);

  const svg = d3.create("svg")
    .attr("viewBox", [0, 0, width, height])
    .style("font-family", "sans-serif");

  const leaf = svg.selectAll("g")
    .data(root.leaves())
    .join("g")
    .attr("transform", d => `translate(${d.x0},${d.y0})`);

  leaf.append("rect")
    .attr("width", d => d.x1 - d.x0)
    .attr("height", d => d.y1 - d.y0)
    .attr("fill", d => color(d.data.enfants))
    .attr("stroke", "white")
    .attr("stroke-width", 1.5)
    .append("title")
    .text(d => `${d.data.enfants}: ${d.data.total_groupe} (${d.data.part_groupe.toFixed(1)}%)`);

  // Étiquettes 
  leaf.append("text")
    .attr("x", 6)
    .attr("y", 18)
    .attr("fill", d => d.data.enfants === "Aucun enfant" || d.data.enfants === "1 enfant" ? "black" : "white")
    .attr("font-size", "12px")
    .attr("font-weight", "bold")
    .text(d => (d.x1 - d.x0) > 60 && (d.y1 - d.y0) > 20 ? d.data.enfants : "");

  leaf.append("text")
    .attr("x", 6)
    .attr("y", 34)
    .attr("fill", d => d.data.enfants === "Aucun enfant" || d.data.enfants === "1 enfant" ? "black" : "white")
    .attr("font-size", "11px")
    .text(d => (d.x1 - d.x0) > 60 && (d.y1 - d.y0) > 40 ? `${d.data.part_groupe.toFixed(1)}%` : "");

  return svg.node();
}
