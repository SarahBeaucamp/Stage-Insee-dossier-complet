? db.query(`
  SELECT TIME_PERIOD, TAB_MEASURE_LABEL, OBS_VALUE
  FROM read_parquet('${chemin_s3}')
  WHERE GEO_OBJECT_LABEL = '${selection_type_geo}'
    AND GEO = '${selection_geo.GEO}'
    AND ID_TAB = 'REU'
`)

Plot.plot({
  title: "Nombre d'électeurs inscrits sur liste principale par élection",
  marginBottom: 40,
  x: { 
    label: "Année de l'élection", 
    tickFormat: "d" // Empêche d'afficher les années avec des virgules (ex: 2,022)
  },
  y: { 
    label: "Nombre d'électeurs", 
    grid: true 
  },
  marks: [
    // 1. Trace une ligne bleue continue entre toutes les années
    Plot.line(elec, { 
      x: "TIME_PERIOD", 
      y: "OBS_VALUE", 
      stroke: "blue", 
      strokeWidth: 2 
    }),
    
    // 2. Ajoute les points avec une info-bulle au survol
    Plot.dot(elec, { 
      x: "TIME_PERIOD", 
      y: "OBS_VALUE", 
      fill: "#174695", 
      r: 4,
      // Ajout interactif : affiche les infos quand on passe la souris sur le point
      title: (d) => `${d.TAB_MEASURE_LABEL} (${d.TIME_PERIOD})\nÉlecteurs : ${d.OBS_VALUE}`
    })
  ]
})
elec_pour_tableau = {
  return elec.slice().sort((a, b) => d3.ascending(a.TIME_PERIOD, b.TIME_PERIOD));
}
