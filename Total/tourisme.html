function classer_hotels(label) {
  if (!label) return null;
  let txt = label.toLowerCase();
  
  if (!txt.includes("hôtels") && !txt.includes("hébergement")) return null;

  if (txt.includes("1 étoile")) return "1 étoile";
  if (txt.includes("2 étoiles")) return "2 étoiles";
  if (txt.includes("3 étoiles")) return "3 étoiles";
  if (txt.includes("4 étoiles")) return "4 étoiles";
  if (txt.includes("5 étoiles")) return "5 étoiles";
  if (txt.includes("non classé")) return "Non classé";
  
  return null;
}

equip_mutate1 = equip_brut1.map(function(d) {
  return {
    OBS_VALUE: d.OBS_VALUE,
    etoiles: classer_hotels(d.TAB_MEASURE_LABEL)
  };
}).filter(d => d.etoiles !== null);

equip_groupes1 = d3.rollups(
  equip_mutate1,
  function(v) { return d3.sum(v, function(d) { return d.OBS_VALUE; }); },
  function(d) { return d.etoiles; }
);

equip_summarise1 = equip_groupes1.map(function(g) {
  return { etoiles: g[0], total_groupe: g[1] };
});

total1 = d3.sum(equip_summarise1, function(d) { return d.total_groupe; });

equip1 = equip_summarise1.map(function(d) {
  return {
    etoiles: d.etoiles,
    total_groupe: d.total_groupe,
    total1: total1,
    part_groupe: total1 > 0 ? (d.total_groupe / total1) * 100 : 0
  };
});
