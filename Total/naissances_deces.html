db = DuckDBClient.of({})

// 4. Définir l'adresse du fichier sur le S3 du SSP Cloud
chemin_s3 = 'https://minio.lab.sspcloud.fr/sarahbeaucamp/dossier_complet.parquet'
isGeoSelected = (selection_geo !== null);
htl.html`<strong>Territoire sélectionné :</strong> ${isGeoSelected ? selection_geo.GEO_LABEL : ""} (${isGeoSelected ? selection_geo.GEO : ""})`
solde_brut = isGeoSelected
? db.query(`
  SELECT
    TAB_MEASURE_LABEL,
    TIME_PERIOD,
    CAST(REPLACE(CAST(OBS_VALUE AS VARCHAR), ',', '.') AS DOUBLE) AS OBS_VALUE
  FROM read_parquet('${chemin_s3}')
  WHERE GEO_OBJECT_LABEL = '${selection_type_geo}'
    AND GEO = '${selection_geo.GEO}'
    AND ID_TAB = 'RFD_G1'
    AND TAB_MEASURE_LABEL IN ('Nombre de naissances vivantes', 'Nombre de décès')
  ORDER BY TIME_PERIOD
`)
: [];

// Équivalent de : group_by(TAB_MEASURE_LABEL, TIME_PERIOD) %>% summarise(total_groupe = sum(OBS_VALUE, na.rm = TRUE))
solde_groupes = d3.rollups(
  solde_brut,
  function(v) { return d3.sum(v, function(d) { return d.OBS_VALUE; }); },
  function(d) { return d.TAB_MEASURE_LABEL; },
  function(d) { return d.TIME_PERIOD; }
);

solde_summarise = solde_groupes.flatMap(function(g) {
  let label = g[0];
  return g[1].map(function(g2) {
    return { TAB_MEASURE_LABEL: label, TIME_PERIOD: g2[0], total_groupe: g2[1] };
  });
});

// Équivalent de : mutate(total = sum(total_groupe), part_groupe = (total_groupe / total) * 100)
total_solde = d3.sum(solde_summarise, function(d) { return d.total_groupe; });

solde = solde_summarise.map(function(d) {
  return {
    TAB_MEASURE_LABEL: d.TAB_MEASURE_LABEL,
    TIME_PERIOD: d.TIME_PERIOD,
    total_groupe: d.total_groupe,
    total: total_solde,
    part_groupe: total_solde > 0 ? (d.total_groupe / total_solde) * 100 : 0
  };
}).sort((a, b) => d3.ascending(+a.TIME_PERIOD, +b.TIME_PERIOD));
// Équivalent du ggplot: geom_line + scale_color_manual + theme_minimal
Plot.plot({
  title: isGeoSelected ? `Naissances et décès domiciliés – ${selection_geo.GEO_LABEL}` : "Sélectionnez un territoire",
  marginBottom: 40,
  marginLeft: 50,
  color: {
    domain: ["Nombre de naissances vivantes", "Nombre de décès"],
    range: ["#174695", "#FB5A5A"],
    legend: true
  },
  x: { 
    label: null, 
    tickFormat: "d",
    grid: false
  },
  y: { 
    label: "en nombre", 
    grid: true 
  },
  marks: [
    Plot.lineY(solde, { x: d => +d.TIME_PERIOD, y: "total_groupe", stroke: "TAB_MEASURE_LABEL", strokeWidth: 2 }),
    Plot.dot(solde, { x: d => +d.TIME_PERIOD, y: "total_groupe", fill: "TAB_MEASURE_LABEL" })
  ]
})
Inputs.table(solde, {

    total_groupe: "Nombre",
    part_groupe: "Part (%)"
  }
})
