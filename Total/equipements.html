equip_filtre1 = isGeoSelected
? db.query(`
  SELECT TAB_MEASURE_LABEL, OBS_VALUE
  FROM read_parquet('${chemin_s3}')
  WHERE TIME_PERIOD = '${selection_annee}'
    AND GEO_OBJECT_LABEL = '${selection_type_geo}'
    AND GEO = '${selection_geo.GEO}'
    AND ID_TAB = 'EQUIP_T1'
    AND TAB_MEASURE_LABEL IN (
      'Nombre d''équipements – Station-service',
      'Nombre d''équipements – Station de recharge de véhicules électriques',
      'Nombre d''équipements – Supermarché et magasin multi-commerce',
      'Nombre d''équipements – Boulangerie-pâtisserie',
      'Nombre d''équipements – Supérette',
      'Nombre d''équipements – Épicerie',
      'Nombre d''équipements – Hypermarché et grand magasin',
      'Nombre d''équipements – Coiffure'
    )
`)
: [];

function case_when_equipements(TAB_MEASURE_LABEL) {
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Station-service") return "Station-service";
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Station de recharge de véhicules électriques") return "Station de recharge de véhicules électriques";
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Boulangerie-pâtisserie") return "Boulangerie-pâtisserie";
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Supérette" || TAB_MEASURE_LABEL === "Nombre d'équipements – Épicerie") return "Epicerie, Supérette";
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Hypermarché et grand magasin" || TAB_MEASURE_LABEL === "Nombre d'équipements – Supermarché et magasin multi-commerce") return "Grande Surface";
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Coiffure") return "Coiffure";
  return null;
}

equip_mutate1 = equip_filtre1.map(function(d) {
  return {
    TAB_MEASURE_LABEL: d.TAB_MEASURE_LABEL,
    OBS_VALUE: d.OBS_VALUE,
    équipements: case_when_equipements(d.TAB_MEASURE_LABEL)
  };
});

equip_groupes1 = d3.rollups(
  equip_mutate1,
  function(v) { return d3.sum(v, function(d) { return d.OBS_VALUE; }); },
  function(d) { return d.équipements; }
);

equip_summarise1 = equip_groupes1.map(function(g) {
  return { équipements: g[0], total_groupe: g[1] };
});

total1 = d3.sum(equip_summarise1, function(d) { return d.total_groupe; });

equip1 = equip_summarise1.map(function(d) {
  return {
    équipements: d.équipements,
    total_groupe: d.total_groupe,
    total1: total1,
    part_groupe: total1 > 0 ? (d.total_groupe / total1) * 100 : 0
  };
});

  let data = equip1.slice().sort((a, b) => d3.descending(a.total_groupe, b.total_groupe));

function classer_equipements3(label) {
  if (!label) return null;
  let txt = label.toLowerCase();

  if (txt.includes("bibliothèque")) return "Bibliothèque";
  if (txt.includes("collège")) return "Collège";
  if (txt.includes("maternelle") || txt.includes("primaire") || txt.includes("élémentaire")) return "École";
  if (txt.includes("général") || txt.includes("technologique")) return "Lycée d'enseignement général et/ou technologique";
  if (txt.includes("professionnel") || txt.includes("agricole")) return "Lycée d'enseignement professionnel ou agricole";
  
  return null; 
}
