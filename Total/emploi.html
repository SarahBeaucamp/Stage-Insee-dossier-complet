// --- Fonction de normalisation : enlève accents, harmonise apostrophes, met en minuscule ---
// Sert à rendre le matching des libellés robuste aux petites variations de texte
// (accents, apostrophes typographiques vs droites, casse...).
function normalize(s) {
  if (!s) return "";
  return s
    .toLowerCase()
    .replace(/[’‘]/g, "'")
    .normalize("NFD")
    .replace(/[\u0300-\u036f]/g, ""); // supprime les accents
}

// 2. Classification corrigée et robuste.
// Important : dans TAB_MEASURE_LABEL, il n'y a pas de catégorie "sans emploi" isolée.
// - "Actif x CSP"          = TOTAL des actifs de cette CSP (occupés + non-occupés)
// - "Actif occupé x CSP"   = uniquement les actifs OCCUPÉS de cette CSP
// Le nombre de personnes "sans emploi" se déduit donc par soustraction : Total - Occupé.
function classer_pop(label) {
  if (!label) return null;
  let txt = normalize(label);

  // Identifier le statut (Total vs Occupé)
  // On teste "actif occupe" avant "actif" seul pour ne pas les confondre.
  let act = null;
  if (txt.includes("actif occupe")) act = "Occupé";
  else if (txt.includes("actif")) act = "Total";
  if (!act) return null;

  // Identifier la CSP
  let csp = null;
  if (txt.includes("agriculteurs")) csp = "Agriculteurs";
  else if (txt.includes("artisans")) csp = "Artisans, commerçants et chefs d’entreprise";
  else if (txt.includes("cadres et professions")) csp = "Cadres et professions intellectuelles supérieures";
  else if (txt.includes("professions intermediaires")) csp = "Professions intermédiaires";
  else if (txt.includes("employes")) csp = "Employés";
  else if (txt.includes("ouvriers")) csp = "Ouvriers";

  if (!csp) return null;

  return { act: act, csp: csp };
}

// 3. Filtrage et nettoyage
equip_mutate = equip_brut.map(function(d) {
  let classification = classer_pop(d.TAB_MEASURE_LABEL);
  if (!classification) return null;

  return {
    OBS_VALUE: Number(d.OBS_VALUE), // sécurité : force un Number même si DuckDB renvoie un BigInt
    ACT: classification.act,
    CSP: classification.csp
  };
}).filter(d => d !== null);

// 4. Agrégation : somme des valeurs par CSP x (Total / Occupé)
// NOTE : on n'utilise volontairement PAS d3.flatRollup ici, car cette fonction
// (ajoutée récemment à d3-array) peut être absente ou indisponible selon la version
// de d3 embarquée par le moteur OJS de Quarto, ce qui ferait planter la cellule
// silencieusement et laisserait tous les graphes vides sans erreur visible.
// On fait donc l'agrégation "à la main" avec une Map, ce qui ne dépend que du JS de base.
csp_map = {
  const m = new Map();
  for (const d of equip_mutate) {
    if (!m.has(d.CSP)) m.set(d.CSP, {});
    const vals = m.get(d.CSP);
    vals[d.ACT] = (vals[d.ACT] || 0) + d.OBS_VALUE;
  }
  return m;
}

// Ordre pour l'affichage
ordre_csp = ["Agriculteurs", "Artisans, commerçants et chefs d’entreprise", "Ouvriers", "Employés", "Professions intermédiaires", "Cadres et professions intellectuelles supérieures"];

// 6. Calcul des parts locales : part d'occupés / sans emploi AU SEIN de chaque CSP
// (Avec Emploi + Sans Emploi = 100% pour chaque CSP)
// On garde aussi population_exacte pour l'histogramme empilé en effectifs.
equip = Array.from(csp_map, ([csp, vals]) => {
  let total = vals["Total"] || 0;
  let occupe = vals["Occupé"] || 0;
  let sans_emploi = total - occupe;

  return [
    {
      CSP: csp,
      ACT: "Avec Emploi",
      population_exacte: occupe,
      part_locale: total > 0 ? occupe / total : 0,
      pourcentage_affichage: total > 0 ? (occupe / total * 100).toFixed(1) + " %" : "0.0 %"
    },
    {
      CSP: csp,
      ACT: "Sans Emploi",
      population_exacte: sans_emploi,
      part_locale: total > 0 ? sans_emploi / total : 0,
      pourcentage_affichage: total > 0 ? (sans_emploi / total * 100).toFixed(1) + " %" : "0.0 %"
    }
  ];
}).flat();

equip_trie = equip.sort((a, b) => ordre_csp.indexOf(a.CSP) - ordre_csp.indexOf(b.CSP));
