function classer_societes(label) {
  if (!label) return null;
  let txt = label.toLowerCase();
  if (txt.includes("société par actions simplifiée")) return "Société par actions simplifiée";
  if (txt.includes("société à responsabilité limitée")) return "Société à responsabilité limitée";
  if (txt.includes("entrepreneur individuel")) return "Entrepreneur individuel";
  if (txt.includes("ensemble")) return "Ensemble";
  if (txt.includes("autres sociétés")) return "Autres sociétés";
  return null;
}

function nettoyer_secteur5(label) {
  if (!label) return null;
  if (!label.includes("Nombre de nouveaux établissements enregistrés – ")) return null;
  return label.replace("Nombre de nouveaux établissements enregistrés – ", "");
}

equip_mutate5 = equip_brut5.map(function(d) {
  return {
    OBS_VALUE: d.OBS_VALUE,
    secteur5: nettoyer_secteur5(d.TAB_MEASURE_LABEL)
  };
}).filter(d => d.secteur5 !== null);

equip_groupes5 = d3.rollups(
  equip_mutate5,
  function(v) { return d3.sum(v, function(d) { return d.OBS_VALUE; }); },
  function(d) { return d.secteur5; }
);

equip_summarise5 = equip_groupes5.map(function(g) {
  return { secteur5: g[0], total_groupe: g[1] };
});

total5 = d3.sum(equip_summarise5, function(d) { return d.total_groupe; });

data_clean5 = equip_summarise5.map(function(d) {
  return {
    secteur5: d.secteur5,
    total_groupe: d.total_groupe,
    total5: total5,
    part_groupe: total5 > 0 ? (d.total_groupe / total5) * 100 : 0
  };
});
