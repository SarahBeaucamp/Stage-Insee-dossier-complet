function classer_pop(label) {
  if (!label) return null;
  let txt = label.toLowerCase();
  
  // Identifier le sexe
  let sexe = null;
  if (txt.includes("homme")) sexe = "Homme";
  else if (txt.includes("femme")) sexe = "Femme";
  if (!sexe) return null; // Si on ne trouve ni homme ni femme, on ignore
  
  let tranche = null;
  if (txt.includes("moins de 15 ans")) tranche = "Moins de 15 ans";
  else if (txt.includes("15 à 24 ans")) tranche = "15-24 ans";
  else if (txt.includes("25 à 39 ans")) tranche = "25-39 ans";
  else if (txt.includes("40 à 54 ans")) tranche = "40-54 ans";
  else if (txt.includes("55 à 64 ans")) tranche = "55-64 ans";
  else if (txt.includes("65 à 79 ans")) tranche = "65-79 ans";
  else if (txt.includes("80 ans")) tranche = "80 ans et plus";
  
  if (!tranche) return null;
  
  return { sexe: sexe, tranche: tranche };
}

equip_mutate = equip_brut.map(function(d) {
  let classification = classer_pop(d.TAB_MEASURE_LABEL);
  if (!classification) return null;
  
  return {
    OBS_VALUE: d.OBS_VALUE,
    SEXE: classification.sexe,
    TRANCHE_AGE: classification.tranche
  };
}).filter(d => d !== null);

equip_groupes = d3.flatRollup(
  equip_mutate,
  v => d3.sum(v, d => d.OBS_VALUE),
  d => d.SEXE,
  d => d.TRANCHE_AGE
).map(([sexe, tranche, valeur]) => ({
  SEXE: sexe, 
  TRANCHE_AGE: tranche, 
  population_exacte: valeur
}));

total_hommes = d3.sum(equip_groupes.filter(d => d.SEXE === "Homme"), d => d.population_exacte);
total_femmes = d3.sum(equip_groupes.filter(d => d.SEXE === "Femme"), d => d.population_exacte);

equip = equip_groupes.map(function(d) {
  let part = 0;
  if (d.SEXE === "Homme" && total_hommes > 0) part = d.population_exacte / total_hommes;
  if (d.SEXE === "Femme" && total_femmes > 0) part = d.population_exacte / total_femmes;
  
  return {
    SEXE: d.SEXE,
    TRANCHE_AGE: d.TRANCHE_AGE,
    population_exacte: d.population_exacte,
    part_locale: part,
    pourcentage_affichage: (part * 100).toFixed(1) + " %"
  };
});

function classer_residence(label) {
  if (!label) return null;
  let txt = label.toLowerCase();
  
  if (!txt.includes("1 an ou plus")) return null;

  if (txt.includes("dans le même logement")) return "Dans le même logement";
  if (txt.includes("autre logement dans la même commune")) return "Dans un autre logement de la même commune";
  
  if (txt.includes("autre commune") || 
      txt.includes("hors de la région") || 
      txt.includes("autre département") || 
      txt.includes("hors métropole")) {
    return "Dans une autre commune";
  }
  
  return null;
}

equip_mutate1 = equip_brut1.map(function(d) {
  return {
    OBS_VALUE: d.OBS_VALUE,
    Lieu_de_résidence: classer_residence(d.TAB_MEASURE_LABEL)
  };
}).filter(d => d.Lieu_de_résidence !== null);

equip_groupes1 = d3.rollups(
  equip_mutate1,
  function(v) { return d3.sum(v, function(d) { return d.OBS_VALUE; }); },

);

equip_summarise1 = equip_groupes1.map(function(g) {

});

total1 = d3.sum(equip_summarise1, function(d) { return d.total_groupe; });

equip1 = equip_summarise1.map(function(d) {

camembert = {
  if (!isGeoSelected || total1 === 0) return htl.html`<div>Sélectionnez un territoire ou aucune donnée disponible pour cette année.</div>`;
  
  const width = 600;
  const height = 350;
  const radius = Math.min(width, height) / 2 - 20;

  const arc = d3.arc().innerRadius(0).outerRadius(radius);
  
  const pie = d3.pie()
    .value(d => d.total_groupe)
    .sort(null)
    .startAngle(2 * Math.PI)
    .endAngle(0);

  const color = d3.scaleOrdinal()
    .domain(ordre_niveaux)
    .range(["#08519c", "#6baed6", "#bdd7e7"]); // Du plus foncé au plus clair

  const svg = d3.create("svg")
    .attr("viewBox", [0, 0, width, height])
    .style("font-family", "sans-serif");

  const gPie = svg.append("g")
    .attr("transform", `translate(${width / 3}, ${height / 2})`);

  gPie.selectAll("path")
    .data(pie(equip_trie1))
    .join("path")
    .attr("fill", d => color(d.data.Lieu_de_résidence))
    .attr("stroke", "white")
    .attr("stroke-width", "1.5px")
    .attr("d", arc)
    .append("title")
    .text(d => `${d.data.Lieu_de_résidence}: ${d.data.total_groupe}`);

  gPie.selectAll("text")
    .data(pie(equip_trie1))
    .join("text")
    .attr("transform", d => `translate(${arc.centroid(d)})`)
    .attr("text-anchor", "middle")
    // Noir pour la couleur claire, blanc pour les couleurs foncées
    .attr("fill", d => d.data.Lieu_de_résidence === "Dans une autre commune" ? "black" : "white") 
    .attr("font-size", "12px")
    .attr("font-weight", "bold")
    // Affiche uniquement si c'est assez grand pour être lisible
    .text(d => d.data.part_groupe >= 3 ? `${d.data.part_groupe.toFixed(1)}%` : "");

  const legend = svg.append("g")
    .attr("transform", `translate(${width * 0.60}, ${height / 2 - 50})`);
    
  legend.append("text")
    .attr("x", 0)
    .attr("y", -10)
    .attr("font-weight", "bold")
    .attr("font-size", "12px")
    .text("Lieu de résidence");

  const categories = color.domain();
  categories.forEach((cat, i) => {
    const row = legend.append("g").attr("transform", `translate(0, ${i * 25 + 10})`);
    row.append("rect").attr("width", 15).attr("height", 15).attr("fill", color(cat)).attr("rx", 2);
    // On ajoute un retour à la ligne manuel en SVG pour les textes longs
    if(cat === "Dans un autre logement de la même commune") {
        row.append("text").attr("x", 25).attr("y", 10).attr("font-size", "12px").text("Dans un autre logement");
        row.append("text").attr("x", 25).attr("y", 24).attr("font-size", "12px").text("de la même commune");
    } else {
        row.append("text").attr("x", 25).attr("y", 12).attr("font-size", "12px").text(cat);
    }
  });

  return svg.node();
}
