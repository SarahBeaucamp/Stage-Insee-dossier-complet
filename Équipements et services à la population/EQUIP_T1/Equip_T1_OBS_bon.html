function case_when_equipements(TAB_MEASURE_LABEL) {
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Station-service") return "Station-service";
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Station de recharge de véhicules électriques") return "Station de recharge de véhicules électriques";
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Boulangerie-pâtisserie") return "Boulangerie-pâtisserie";
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Supérette" || TAB_MEASURE_LABEL === "Nombre d'équipements – Épicerie") return "Epicerie, Supérette";
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Hypermarché et grand magasin" || TAB_MEASURE_LABEL === "Nombre d'équipements – Supermarché et magasin multi-commerce") return "Grande Surface";
  if (TAB_MEASURE_LABEL === "Nombre d'équipements – Coiffure") return "Coiffure";
  return null;
}

equip_mutate = equip_filtre.map(function(d) {
  return {
    TAB_MEASURE_LABEL: d.TAB_MEASURE_LABEL,
    OBS_VALUE: d.OBS_VALUE,
    équipements: case_when_equipements(d.TAB_MEASURE_LABEL)
  };
});

// group_by(équipements) %>% summarise(total_groupe = sum(OBS_VALUE, na.rm = TRUE)) %>% collect()
equip_groupes = d3.rollups(
  equip_mutate,
  function(v) { return d3.sum(v, function(d) { return d.OBS_VALUE; }); },
  function(d) { return d.équipements; }
);

equip_summarise = equip_groupes.map(function(g) {
  return { équipements: g[0], total_groupe: g[1] };
});

// mutate(total = sum(total_groupe), part_groupe = (total_groupe / total) * 100)
total = d3.sum(equip_summarise, function(d) { return d.total_groupe; });

equip = equip_summarise.map(function(d) {
  return {
    équipements: d.équipements,
    total_groupe: d.total_groupe,
    total: total,
    part_groupe: total > 0 ? (d.total_groupe / total) * 100 : 0
  };
});
