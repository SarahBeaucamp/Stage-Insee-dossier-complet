// 2. On classe selon les mêmes correspondances que dans le code R (etab_spheres)
correspondance_etab = new Map([
  ["Établissements – Ensemble hors particuliers employeurs x Établissements appartenant à la sphère productive", "Sphère productive"],
  ["Établissements – Fonction Publique x Établissements appartenant à la sphère présentielle", "Sphère présentielle, domaine public"],
  ["Établissements – Ensemble hors particuliers employeurs x Établissements appartenant à la sphère présentielle", "Sphère présentielle"],
  ["Établissements – Fonction Publique x Établissements appartenant à la sphère productive", "Sphère productive, domaine public"]
]);

function classer_etab(label) {
  if (!label) return null;
  return correspondance_etab.has(label) ? correspondance_etab.get(label) : null;
}

// 3. Filtrage et nettoyage
etab_mutate = etab_brut.map(function(d) {
  return {
    OBS_VALUE: d.OBS_VALUE,
    type_etab: classer_etab(d.TAB_MEASURE_LABEL)
  };
}).filter(d => d.type_etab !== null);

// 4. Agrégation des résultats (équivalent group_by + summarise)
etab_groupes = d3.rollups(
  etab_mutate,
  function(v) { return d3.sum(v, function(d) { return d.OBS_VALUE; }); },
  function(d) { return d.type_etab; }
);

etab_summarise = etab_groupes.map(function(g) {
  return { type_etab: g[0], total_groupe: g[1] };
});

total_etab = d3.sum(etab_summarise, function(d) { return d.total_groupe; });

etab_spheres = etab_summarise.map(function(d) {
  return {
    type_etab: d.type_etab,
    total_groupe: d.total_groupe,
    total: total_etab,
    part_groupe: total_etab > 0 ? (d.total_groupe / total_etab) * 100 : 0
  };
});

// 5. Ordre logique des catégories pour l'affichage (ordre du code R)
ordre_etab = Array.from(correspondance_etab.values());

etab_trie = etab_spheres.sort((a, b) => ordre_etab.indexOf(a.type_etab) - ordre_etab.indexOf(b.type_etab));
// Histogramme (équivalent ggplot2 : geom_col + labels tronqués)
Plot.plot({
  title: isGeoSelected ? `Établissements selon les sphères de l'économie fin ${selection_annee} – ${selection_geo.GEO_LABEL}` : "Sélectionnez un territoire",
  marginBottom: 100,
  x: { 
    domain: ordre_etab, // Force l'ordre sur l'axe X
    label: null,
    tickFormat: (d) => d.length > 18 ? d.slice(0, 18) + "…" : d
  },
  y: { 
    label: "Nombre d'établissements", 
    grid: true 
  },
  marks: [
    Plot.barY(etab_trie, { x: "type_etab", y: "total_groupe", fill: "steelblue" }),
    Plot.ruleY([0])
  ]
})
etab_pour_tableau = {
  let data = etab_trie.slice();
  data.push({ type_etab: "TOTAL", total_groupe: total_etab });
  return data;
}

// Affichage du tableau

  header: {
    type_etab: "Sphère",
    total_groupe: "Nombre d'établissements"
  }
